def baseline_model(image):
    '''
    This is the baseline predictive model that takes in the image and returns a label prediction
    '''
    feature1 = np.histogram(image[:,:,0],bins=bins)[0]
    feature2 = np.histogram(image[:,:,1],bins=bins)[0]
    feature3 = np.histogram(image[:,:,2],bins=bins)[0]
    feature = np.concatenate((feature1, feature2, feature3), axis=None).reshape(1,-1)
    return clf.predict(feature)

class base_CNN(nn.Module):
    def __init__(self):
        super(base_CNN, self).__init__()
        self.conv1 = nn.Conv2d(in_channels=3, out_channels=32, kernel_size=3,stride=1,padding_mode='zeros')
        self.conv2 = nn.Conv2d(in_channels=32, out_channels=64, kernel_size=3,stride=1,padding_mode='zeros')
        self.pool = nn.MaxPool2d(2, 2)

    def forward(self, x):
        x = self.pool(F.relu(self.conv1(x)))
        x = self.pool(F.relu(self.conv2(x)))
        x = torch.flatten(x, 1) # flatten all dimensions except batch
        return x
    
#Label_Clean is a model built based on base_CNN and take 'image' and 'noisy_label' as input, clean_label as part of loss function
class Label_Clean(nn.Module):
    def __init__(self,base_CNN):
        super(Label_Clean, self).__init__()
        self.base_CNN=base_CNN
        self.fc11 = nn.LazyLinear(out_features=32)
        self.fc12 = nn.LazyLinear(out_features=32)
        self.fc2 = nn.LazyLinear(out_features=10)
        
    def forward(self,x1,x2):
        #x1 takes the image, x2 takes the noisy label input
        noisy=x2
        x1=self.base_CNN(x1)
        x1=self.fc11(x1)
        x2=self.fc12(x2)
        x = torch.cat([x1,x2],dim=1)
        x = self.fc2(x)
        x = x+noisy
        return x

#Build Customized Dataset
class Cleandata(Dataset):
    def __init__(self, image ,noisy, clean):
        self.image=image
        self.clean=clean
        self.noisy=noisy

    def __len__(self):
        return len(self.image)

    def __getitem__(self, idx):
        X=self.image[idx,]
        n=self.noisy[idx,]
        c=self.clean[idx,]
        return(X,n,c)
